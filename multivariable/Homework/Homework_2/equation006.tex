\begin{vmatrix}
\overrightarrow{i} & \overrightarrow{j} & \overrightarrow{k} \\
1 & 1 & 0 \\
0 & 1 & 1
\end{vmatrix} = \\

\overrightarrow{i} (1 \cdot 1 - 1 \cdot 0) -
\overrightarrow{j} (1 \cdot 1 - 0 \cdot 0) +
\overrightarrow{k} (1 \cdot 1 - 0 \cdot 1) = \\

\overrightarrow{i} \cdot 1 - \overrightarrow{j} \cdot 1 + \overrightarrow{k} \cdot 1 = \\

\begin{bmatrix}
1 \\ 0 \\ 0
\end{bmatrix}
-
\begin{bmatrix}
0 \\ 1 \\ 0
\end{bmatrix}
+
\begin{bmatrix}
0 \\ 0 \\ 1
\end{bmatrix} = \\
\begin{bmatrix}
1 \\ -1 \\ 1
\end{bmatrix}
