<
\left(\frac{\partial f}{\partial x_1}, ..., \frac{\partial f}{\partial x_n}\right),
\left(\frac{\partial f}{\partial x_1}, ..., \frac{\partial f}{\partial x_n}\right)
>
=\\
\frac{\partial f}{\partial x_1} \cdot \frac{\partial f}{\partial x_1} +
... +
\frac{\partial f}{\partial x_n} \cdot \frac{\partial f}{\partial x_n}
=\\
\frac{\partial^2 f}{\partial x_1^2} +
... +
\frac{\partial^2 f}{\partial x_n^2}
